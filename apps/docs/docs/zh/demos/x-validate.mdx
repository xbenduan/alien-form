# x-validate 动态校验

演示 `x-validate` 动态校验：表达式校验、跨字段比较、异步 handler 校验。

```tsx preview
import { createForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

const schema = {
  type: "object",
  properties: {
    username: {
      type: "string",
      title: "用户名",
      default: "guest",
      component: "Input",
      decorator: "FormItem",
      props: { placeholder: "请输入 admin" },
      "x-validate": {
        type: "expression",
        expression: "$value === 'admin' ? undefined : '用户名必须是 admin'",
      },
    },
    password: {
      type: "string",
      title: "密码",
      component: "Input",
      decorator: "FormItem",
      props: { type: "password", placeholder: "至少 6 位" },
      validators: [
        { required: true, message: "请输入密码" },
        { minLength: 6, message: "密码至少 6 位" },
      ],
    },
    confirmPassword: {
      type: "string",
      title: "确认密码",
      component: "Input",
      decorator: "FormItem",
      props: { type: "password", placeholder: "再次输入密码" },
      "x-validate": {
        type: "expression",
        dependencies: { password: "password" },
        expression: '$value === $deps.password ? undefined : "两次输入的密码不一致"',
      },
    },
  },
};

const components = { Input: SchemaInput, Textarea: SchemaTextarea, Select: SchemaSelect, Checkbox, Switch, DateInput, RadioGroup, Rating, FormGrid, FormLayout, FormSection, ArrayCards, ArrayTable };
const decorators = { FormItem };

export default () => {
  const form = createForm();
  return (
    <FormProvider form={form} components={components} decorators={decorators}>
      <SchemaField schema={schema} />
    </FormProvider>
  );
};
```
