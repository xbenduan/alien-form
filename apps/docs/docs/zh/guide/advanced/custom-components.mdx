# 自定义组件

AlienForm 的组件体系分三层：**普通组件**（编辑单值）、**高阶组件**（组织子字段）、**Decorator**（包装外壳）。

## 普通组件

遵循 `value / onChange` 协议，编辑单个字段值。视觉可以复杂，但最终只通过 `onChange` 返回一个值。

```tsx preview
import React from "react";
import { useCreateForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

// 自定义普通组件：带图片预览的 URL 输入
const ImageInput = ({ value, onChange, placeholder, disabled }) => {
  const url = value ?? "";
  return (
    <div className="flex items-start gap-3">
      <div className="h-16 w-16 shrink-0 overflow-hidden rounded-md border bg-muted flex items-center justify-center">
        {url ? (
          <img src={url} alt="preview" className="h-full w-full object-cover" />
        ) : (
          <span className="text-[10px] text-muted-foreground">No image</span>
        )}
      </div>
      <input
        className="flex-1 h-9 rounded-md border border-input bg-background px-3 py-1 text-sm"
        value={url}
        onChange={(e) => onChange?.(e.target.value)}
        placeholder={placeholder}
        disabled={disabled}
      />
    </div>
  );
};

const schema = {
  type: "object",
  properties: {
    name: {
      type: "string",
      title: "商品名称",
      component: "Input",
      decorator: "FormItem",
      validators: [{ required: true, message: "必填" }],
    },
    image: {
      type: "string",
      title: "商品图片",
      component: "ImageInput",
      decorator: "FormItem",
      default: "https://picsum.photos/200",
      props: { placeholder: "输入图片 URL" },
    },
    score: {
      type: "number",
      title: "评分",
      component: "Rating",
      decorator: "FormItem",
      default: 3,
    },
  },
};

const components = { Input: SchemaInput, Textarea: SchemaTextarea, Select: SchemaSelect, Checkbox, Switch, DateInput, RadioGroup, Rating, ImageInput, FormGrid, FormLayout, FormSection, ArrayCards, ArrayTable };
const decorators = { FormItem };

export default () => {
  const [result, setResult] = React.useState<any>(null);
  const form = useCreateForm();

  const handleSubmit = async () => {
    try {
      const values = await form.submit();
      setResult({ success: true, values });
    } catch (err: any) {
      setResult({ success: false, errors: err?.messages || [String(err)] });
    }
  };

  return (
    <div className="space-y-4">
      <FormProvider form={form} components={components} decorators={decorators}>
        <SchemaField schema={schema} />
      </FormProvider>
      <div className="flex gap-2 border-t pt-4">
        <button onClick={handleSubmit} className="rounded-md bg-primary px-4 py-2 text-sm text-primary-foreground hover:bg-primary/90">提交</button>
        <button onClick={() => { form.reset(); setResult(null); }} className="rounded-md border px-4 py-2 text-sm hover:bg-accent">重置</button>
      </div>
      {result && (
        <pre className="rounded-lg bg-muted p-4 text-sm overflow-auto max-h-[300px]">
          <code>{JSON.stringify(result, null, 2)}</code>
        </pre>
      )}
    </div>
  );
};
```

## 高阶组件：分组容器（void）

`type: "void"` 节点自身**不产出值**，子字段路径透明。适合卡片、栅格、折叠面板等纯布局。

```tsx preview
import React from "react";
import { useCreateForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

const schema = {
  type: "object",
  properties: {
    basic: {
      type: "void",
      title: "基本信息",
      component: "FormSection",
      props: { bordered: true, collapsible: true },
      properties: {
        name: { type: "string", title: "姓名", component: "Input", decorator: "FormItem", validators: [{ required: true, message: "必填" }] },
        age: { type: "number", title: "年龄", component: "Input", decorator: "FormItem", props: { type: "number" } },
      },
    },
    grid: {
      type: "void",
      title: "联系方式（栅格布局）",
      component: "FormGrid",
      props: { columns: 2 },
      properties: {
        email: { type: "string", title: "邮箱", component: "Input", decorator: "FormItem" },
        phone: { type: "string", title: "手机", component: "Input", decorator: "FormItem" },
      },
    },
  },
};

const components = { Input: SchemaInput, Textarea: SchemaTextarea, Select: SchemaSelect, Checkbox, Switch, DateInput, RadioGroup, Rating, FormGrid, FormLayout, FormSection, ArrayCards, ArrayTable };
const decorators = { FormItem };

export default () => {
  const [result, setResult] = React.useState<any>(null);
  const form = useCreateForm();

  const handleSubmit = async () => {
    try {
      const values = await form.submit();
      setResult({ success: true, values });
    } catch (err: any) {
      setResult({ success: false, errors: err?.messages || [String(err)] });
    }
  };

  return (
    <div className="space-y-4">
      <FormProvider form={form} components={components} decorators={decorators}>
        <SchemaField schema={schema} />
      </FormProvider>
      <div className="flex gap-2 border-t pt-4">
        <button onClick={handleSubmit} className="rounded-md bg-primary px-4 py-2 text-sm text-primary-foreground hover:bg-primary/90">提交</button>
        <button onClick={() => { form.reset(); setResult(null); }} className="rounded-md border px-4 py-2 text-sm hover:bg-accent">重置</button>
      </div>
      {result && (
        <pre className="rounded-lg bg-muted p-4 text-sm overflow-auto max-h-[300px]">
          <code>{JSON.stringify(result, null, 2)}</code>
        </pre>
      )}
    </div>
  );
};
```

> 提交结果中 `name`、`age`、`email`、`phone` 都在顶层——void 节点 `basic` / `grid` 不贡献路径。

## 高阶组件：数组容器（array）

`type: "array"` + `items` 定义可增删行的列表。每行展开 `items.properties` 中的子字段。

```tsx preview

import { useCreateForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

const schema = {
  type: "object",
  properties: {
    contacts: {
      type: "array",
      title: "联系人列表",
      component: "ArrayCards",
      decorator: "FormItem",
      props: { addText: "+ 添加联系人", maxItems: 5 },
      items: {
        type: "object",
        properties: {
          name: { type: "string", title: "姓名", component: "Input", decorator: "FormItem", validators: [{ required: true, message: "必填" }] },
          phone: { type: "string", title: "电话", component: "Input", decorator: "FormItem" },
          role: {
            type: "string",
            title: "角色",
            component: "Select",
            decorator: "FormItem",
            dataSource: [
              { label: "主要联系人", value: "primary" },
              { label: "备选联系人", value: "secondary" },
            ],
          },
        },
      },
    },

```

## 自定义 Decorator

Decorator 负责字段外壳：标签、必填标记、校验错误、描述。通过 `decorators` 注册，Schema 中用 `"decorator"` 引用。

```tsx preview
import React from "react";
import { useCreateForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

// 自定义 Decorator：水平排列标签和输入
const InlineFormItem = ({ label, required, errors = [], description, children }) => (
  <div className="flex items-start gap-3 mb-3">
    {label && (
      <label className="w-24 pt-2 text-sm text-right shrink-0 text-muted-foreground">
        {required && <span className="text-destructive">*</span>}
        {label}
      </label>
    )}
    <div className="flex-1">
      {children}
      {description && errors.length === 0 && (
        <p className="text-xs text-muted-foreground mt-1">{description}</p>
      )}
      {errors.length > 0 && (
        <p className="text-xs text-destructive mt-1">{errors.map((e) => e.message).join(", ")}</p>
      )}
    </div>
  </div>
);

const schema = {
  type: "object",
  properties: {
    username: {
      type: "string",
      title: "用户名",
      component: "Input",
      decorator: "InlineFormItem",
      validators: [{ required: true, message: "必填" }, { minLength: 3, message: "至少 3 个字符" }],
    },
    email: {
      type: "string",
      title: "邮箱",
      component: "Input",
      decorator: "InlineFormItem",
      description: "用于接收通知",
      validators: [{ required: true, message: "必填" }],
    },
    bio: {
      type: "string",
      title: "个人简介",
      component: "Textarea",
      decorator: "InlineFormItem",
      props: { rows: 3 },
    },
    defaultItem: {
      type: "string",
      title: "默认 FormItem",
      component: "Input",
      decorator: "FormItem",
      description: "对比：这是默认竖向 FormItem",
    },
  },
};

const components = { Input: SchemaInput, Textarea: SchemaTextarea, Select: SchemaSelect, Checkbox, Switch, DateInput, RadioGroup, Rating, FormGrid, FormLayout, FormSection, ArrayCards, ArrayTable };
const decorators = { FormItem, InlineFormItem };

export default () => {
  const [result, setResult] = React.useState<any>(null);
  const form = useCreateForm();

  const handleSubmit = async () => {
    try {
      const values = await form.submit();
      setResult({ success: true, values });
    } catch (err: any) {
      setResult({ success: false, errors: err?.messages || [String(err)] });
    }
  };

  return (
    <div className="space-y-4">
      <FormProvider form={form} components={components} decorators={decorators}>
        <SchemaField schema={schema} />
      </FormProvider>
      <div className="flex gap-2 border-t pt-4">
        <button onClick={handleSubmit} className="rounded-md bg-primary px-4 py-2 text-sm text-primary-foreground hover:bg-primary/90">提交</button>
        <button onClick={() => { form.reset(); setResult(null); }} className="rounded-md border px-4 py-2 text-sm hover:bg-accent">重置</button>
      </div>
      {result && (
        <pre className="rounded-lg bg-muted p-4 text-sm overflow-auto max-h-[300px]">
          <code>{JSON.stringify(result, null, 2)}</code>
        </pre>
      )}
    </div>
  );
};
```

## 选择决策

| 你需要… | 选择 |
| --- | --- |
| 编辑单个字段值（文本/图片/评分/颜色） | 普通组件 `value/onChange` |
| 纯 UI 分组（卡片/栅格/折叠） | `void` + 高阶组件 |
| 子字段聚合为对象 | `object` + 高阶组件 |
| 可增删行的列表 | `array` + 高阶组件 |
| 字段外壳（标签/错误/描述） | Decorator |
