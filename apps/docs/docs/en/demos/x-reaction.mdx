# x-reaction Linkage

Demonstrates `x-reaction` for conditional display, computed values, and dynamic property linking.

```tsx preview
import React from "react";
import { useCreateForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

const schema = {
  type: "object",
  properties: {
    contactType: {
      type: "string",
      title: "联系方式",
      default: "email",
      component: "Select",
      decorator: "FormItem",
      dataSource: [
        { label: "邮箱", value: "email" },
        { label: "电话", value: "phone" },
        { label: "飞书", value: "lark" },
      ],
    },
    email: {
      type: "string",
      title: "邮箱地址",
      component: "Input",
      decorator: "FormItem",
      props: { placeholder: "name@example.com" },
      "x-reaction": {
        display: {
          type: "match",
          dependencies: { contactType: "contactType" },
          match: { email: "visible", default: "none" },
        },
      },
    },
    phone: {
      type: "string",
      title: "手机号",
      component: "Input",
      decorator: "FormItem",
      props: { placeholder: "请输入手机号" },
      state: { display: "none" },
      "x-reaction": {
        display: {
          type: "match",
          dependencies: { contactType: "contactType" },
          match: { phone: "visible", default: "none" },
        },
      },
    },
    larkId: {
      type: "string",
      title: "飞书 ID",
      component: "Input",
      decorator: "FormItem",
      state: { display: "none" },
      "x-reaction": {
        display: {
          type: "match",
          dependencies: { contactType: "contactType" },
          match: { lark: "visible", default: "none" },
        },
      },
    },
    quantity: { type: "number", title: "数量", default: 2, component: "Input", decorator: "FormItem", props: { type: "number" } },
    unitPrice: { type: "number", title: "单价", default: 99, component: "Input", decorator: "FormItem", props: { type: "number" } },
    totalPrice: {
      type: "number",
      title: "总价（联动计算）",
      component: "Input",
      decorator: "FormItem",
      state: { pattern: "readPretty" },
      "x-reaction": {
        value: {
          type: "expression",
          dependencies: { quantity: "quantity", unitPrice: "unitPrice" },
          expression: "($deps.quantity || 0) * ($deps.unitPrice || 0)",
        },
      },
    },
  },
};

const components = { Input: SchemaInput, Textarea: SchemaTextarea, Select: SchemaSelect, Checkbox, Switch, DateInput, RadioGroup, Rating, FormGrid, FormLayout, FormSection, ArrayCards, ArrayTable };
const decorators = { FormItem };

export default () => {
  const [result, setResult] = React.useState<any>(null);
  const form = useCreateForm();

  const handleSubmit = async () => {
    try {
      const values = await form.submit();
      setResult({ success: true, values });
    } catch (err: any) {
      setResult({ success: false, errors: err?.messages || [String(err)] });
    }
  };

  return (
    <div className="space-y-4">
      <FormProvider form={form} components={components} decorators={decorators}>
        <SchemaField schema={schema} />
      </FormProvider>
      <div className="flex gap-2 border-t pt-4">
        <button
          onClick={handleSubmit}
          className="rounded-md bg-primary px-4 py-2 text-sm text-primary-foreground hover:bg-primary/90"
        >
          提交
        </button>
        <button
          onClick={() => { form.reset(); setResult(null); }}
          className="rounded-md border px-4 py-2 text-sm hover:bg-accent"
        >
          重置
        </button>
      </div>
      {result && (
        <pre className="rounded-lg bg-muted p-4 text-sm overflow-auto max-h-[300px]">
          <code>{JSON.stringify(result, null, 2)}</code>
        </pre>
      )}
    </div>
  );
};
```
