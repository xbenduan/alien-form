# State & Display

Demonstrates field state control: `display`, `pattern` (editable/readPretty), conditional visibility via `x-reaction`.

```tsx preview
import React from "react";
import { useCreateForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

const schema = {
  type: "object",
  properties: {
    mode: {
      type: "string",
      title: "表单模式",
      default: "edit",
      component: "RadioGroup",
      decorator: "FormItem",
      dataSource: [
        { label: "编辑", value: "edit" },
        { label: "只读", value: "readonly" },
        { label: "隐藏敏感项", value: "secure" },
      ],
    },
    nickname: {
      type: "string",
      title: "昵称",
      default: "AlienForm",
      component: "Input",
      decorator: "FormItem",
      "x-reaction": {
        pattern: {
          type: "match",
          dependencies: { mode: "mode" },
          match: { readonly: "readPretty", default: "editable" },
        },
      },
    },
    apiSecret: {
      type: "string",
      title: "接口密钥",
      default: "sk_demo_123456",
      component: "Input",
      decorator: "FormItem",
      props: { type: "password" },
      "x-reaction": {
        display: {
          type: "match",
          dependencies: { mode: "mode" },
          match: { secure: "none", default: "visible" },
        },
      },
    },
    reason: {
      type: "string",
      title: "隐藏原因",
      component: "Textarea",
      decorator: "FormItem",
      props: { rows: 3 },
      state: { display: "none" },
      "x-reaction": {
        display: {
          type: "match",
          dependencies: { mode: "mode" },
          match: { secure: "visible", default: "none" },
        },
        value: {
          type: "static",
          value: "secure 模式下敏感字段由外层权限/装饰器范式控制",
        },
      },
    },
  },
};

const components = { Input: SchemaInput, Textarea: SchemaTextarea, Select: SchemaSelect, Checkbox, Switch, DateInput, RadioGroup, Rating, FormGrid, FormLayout, FormSection, ArrayCards, ArrayTable };
const decorators = { FormItem };

export default () => {
  const [result, setResult] = React.useState<any>(null);
  const form = useCreateForm();

  const handleSubmit = async () => {
    try {
      const values = await form.submit();
      setResult({ success: true, values });
    } catch (err: any) {
      setResult({ success: false, errors: err?.messages || [String(err)] });
    }
  };

  return (
    <div className="space-y-4">
      <FormProvider form={form} components={components} decorators={decorators}>
        <SchemaField schema={schema} />
      </FormProvider>
      <div className="flex gap-2 border-t pt-4">
        <button
          onClick={handleSubmit}
          className="rounded-md bg-primary px-4 py-2 text-sm text-primary-foreground hover:bg-primary/90"
        >
          提交
        </button>
        <button
          onClick={() => { form.reset(); setResult(null); }}
          className="rounded-md border px-4 py-2 text-sm hover:bg-accent"
        >
          重置
        </button>
      </div>
      {result && (
        <pre className="rounded-lg bg-muted p-4 text-sm overflow-auto max-h-[300px]">
          <code>{JSON.stringify(result, null, 2)}</code>
        </pre>
      )}
    </div>
  );
};
```
