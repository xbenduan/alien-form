# Layout & Collections

Demonstrates `FormSection`, `FormGrid`, `ArrayCards` for grouping, grid layout, and list input patterns.

```tsx preview
import { createForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

const schema = {
  type: "object",
  properties: {
    basicSection: {
      type: "void",
      title: "基础信息",
      component: "FormSection",
      props: { bordered: true, collapsible: true },
      properties: {
        basicGrid: {
          type: "void",
          component: "FormGrid",
          props: { columns: 2, gap: 4 },
          properties: {
            projectName: {
              type: "string",
              title: "项目名称",
              required: true,
              component: "Input",
              decorator: "FormItem",
              props: { placeholder: "请输入项目名称" },
            },
            projectType: {
              type: "string",
              title: "项目类型",
              default: "internal",
              component: "Select",
              decorator: "FormItem",
              dataSource: [
                { label: "内部平台", value: "internal" },
                { label: "商业化", value: "commercial" },
                { label: "基础设施", value: "infra" },
              ],
            },
            startDate: { type: "string", title: "开始日期", component: "DateInput", decorator: "FormItem" },
            riskLevel: { type: "number", title: "风险评分", default: 3, component: "Rating", decorator: "FormItem", props: { max: 5 } },
          },
        },
      },
    },
    contacts: {
      type: "array",
      title: "项目联系人",
      component: "ArrayCards",
      decorator: "FormItem",
      props: { addText: "+ 添加联系人" },
      items: {
        type: "object",
        properties: {
          name: { type: "string", title: "姓名", required: true, component: "Input", decorator: "FormItem" },
          role: {
            type: "string", title: "角色", component: "Select", decorator: "FormItem",
            dataSource: [{ label: "PM", value: "pm" }, { label: "RD", value: "rd" }, { label: "QA", value: "qa" }],
          },
          email: { type: "string", title: "邮箱", component: "Input", decorator: "FormItem" },
        },
      },
    },
  },
};

const components = { Input: SchemaInput, Textarea: SchemaTextarea, Select: SchemaSelect, Checkbox, Switch, DateInput, RadioGroup, Rating, FormGrid, FormLayout, FormSection, ArrayCards, ArrayTable };
const decorators = { FormItem };

export default () => {
  const form = createForm();
  return (
    <FormProvider form={form} components={components} decorators={decorators}>
      <SchemaField schema={schema} />
    </FormProvider>
  );
};
```
