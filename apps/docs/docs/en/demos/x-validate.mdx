# x-validate Validation

Demonstrates `x-validate` for dynamic validation: expression-based, cross-field comparison, and async handlers.

```tsx preview
import React from "react";
import { useCreateForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

const schema = {
  type: "object",
  properties: {
    username: {
      type: "string",
      title: "用户名",
      default: "guest",
      component: "Input",
      decorator: "FormItem",
      props: { placeholder: "请输入 admin" },
      "x-validate": {
        type: "expression",
        expression: "$value === 'admin' ? undefined : '用户名必须是 admin'",
      },
    },
    password: {
      type: "string",
      title: "密码",
      component: "Input",
      decorator: "FormItem",
      props: { type: "password", placeholder: "至少 6 位" },
      validators: [
        { required: true, message: "请输入密码" },
        { minLength: 6, message: "密码至少 6 位" },
      ],
    },
    confirmPassword: {
      type: "string",
      title: "确认密码",
      component: "Input",
      decorator: "FormItem",
      props: { type: "password", placeholder: "再次输入密码" },
      "x-validate": {
        type: "expression",
        dependencies: { password: "password" },
        expression: '$value === $deps.password ? undefined : "两次输入的密码不一致"',
      },
    },
  },
};

const components = { Input: SchemaInput, Textarea: SchemaTextarea, Select: SchemaSelect, Checkbox, Switch, DateInput, RadioGroup, Rating, FormGrid, FormLayout, FormSection, ArrayCards, ArrayTable };
const decorators = { FormItem };

export default () => {
  const [result, setResult] = React.useState<any>(null);
  const form = useCreateForm();

  const handleSubmit = async () => {
    try {
      const values = await form.submit();
      setResult({ success: true, values });
    } catch (err: any) {
      setResult({ success: false, errors: err?.messages || [String(err)] });
    }
  };

  return (
    <div className="space-y-4">
      <FormProvider form={form} components={components} decorators={decorators}>
        <SchemaField schema={schema} />
      </FormProvider>
      <div className="flex gap-2 border-t pt-4">
        <button
          onClick={handleSubmit}
          className="rounded-md bg-primary px-4 py-2 text-sm text-primary-foreground hover:bg-primary/90"
        >
          提交
        </button>
        <button
          onClick={() => { form.reset(); setResult(null); }}
          className="rounded-md border px-4 py-2 text-sm hover:bg-accent"
        >
          重置
        </button>
      </div>
      {result && (
        <pre className="rounded-lg bg-muted p-4 text-sm overflow-auto max-h-[300px]">
          <code>{JSON.stringify(result, null, 2)}</code>
        </pre>
      )}
    </div>
  );
};
```
