# x-format Transform

Demonstrates `x-format` for input/output value transformation: cents↔yuan, trim+uppercase, enum mapping.

```tsx preview
import React from "react";
import { useCreateForm, FormProvider, SchemaField } from "@alien-form/react";
import {
  SchemaInput,
  SchemaTextarea,
  SchemaSelect,
  Checkbox,
  Switch,
  DateInput,
  RadioGroup,
  Rating,
  FormItem,
  FormGrid,
  FormLayout,
  FormSection,
  ArrayCards,
  ArrayTable,
} from "@alien-form/ui";

const schema = {
  type: "object",
  properties: {
    amount: {
      type: "number",
      title: "金额（页面展示元，提交输出分）",
      default: 12345,
      component: "Input",
      decorator: "FormItem",
      props: { type: "number", placeholder: "例如：123.45" },
      description: "default 为 12345 分，页面展示 123.45；提交时还原为 12345。",
      "x-format": {
        input: { type: "expression", expression: "$value / 100" },
        output: { type: "expression", expression: "($value || 0) * 100" },
      },
      validators: [{ required: true, message: "请输入金额" }],
    },
    code: {
      type: "string",
      title: "编码（trim + uppercase）",
      default: " bao-001 ",
      component: "Input",
      decorator: "FormItem",
      props: { placeholder: "输入会自动标准化" },
      "x-format": {
        input: { type: "computed", handler: "normalizeCode" },
        output: { type: "computed", handler: "normalizeCode" },
      },
    },
    status: {
      type: "string",
      title: "状态映射",
      default: 1,
      component: "Select",
      decorator: "FormItem",
      dataSource: [
        { label: "启用", value: "enabled" },
        { label: "禁用", value: "disabled" },
      ],
      "x-format": {
        input: { type: "match", match: { "1": "enabled", "0": "disabled", default: "disabled" } },
        output: { type: "match", match: { enabled: 1, disabled: 0, default: 0 } },
      },
    },
  },
};

const components = { Input: SchemaInput, Textarea: SchemaTextarea, Select: SchemaSelect, Checkbox, Switch, DateInput, RadioGroup, Rating, FormGrid, FormLayout, FormSection, ArrayCards, ArrayTable };
const decorators = { FormItem };

export default () => {
  const [result, setResult] = React.useState<any>(null);
  const form = useCreateForm({
    handlers: {
      normalizeCode: ({ value }) => (typeof value === "string" ? value.trim().toUpperCase() : value),
    },
  });

  const handleSubmit = async () => {
    try {
      const values = await form.submit();
      setResult({ success: true, values });
    } catch (err: any) {
      setResult({ success: false, errors: err?.messages || [String(err)] });
    }
  };

  return (
    <div className="space-y-4">
      <FormProvider form={form} components={components} decorators={decorators}>
        <SchemaField schema={schema} />
      </FormProvider>
      <div className="flex gap-2 border-t pt-4">
        <button
          onClick={handleSubmit}
          className="rounded-md bg-primary px-4 py-2 text-sm text-primary-foreground hover:bg-primary/90"
        >
          提交
        </button>
        <button
          onClick={() => { form.reset(); setResult(null); }}
          className="rounded-md border px-4 py-2 text-sm hover:bg-accent"
        >
          重置
        </button>
      </div>
      {result && (
        <pre className="rounded-lg bg-muted p-4 text-sm overflow-auto max-h-[300px]">
          <code>{JSON.stringify(result, null, 2)}</code>
        </pre>
      )}
    </div>
  );
};
```
